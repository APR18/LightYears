cmake_minimum_required(VERSION 3.31.3)

project(LightYears
VERSION 1.0.0
LANGUAGES C CXX
)



set(CMAKE_CXX_STANDARD			14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS		OFF)

# This creates a variable so the value is firstTutorial and variable name is PROJECT_TARGET_NAME

include(FetchContent)

set(SFML_LIB sfml)

FetchContent_Declare(${SFML_LIB}
GIT_REPOSITORY https://github.com/SFML/SFML
GIT_TAG 2.6.0
GIT_SHALLOW FALSE
)

FetchContent_MakeAvailable(${SFML_LIB})
set(PROJECT_TARGET_NAME LightYearsGame)
set(PROJECT_ENGINE_TARGET_NAME LightYearsEngine)

add_subdirectory(LightYearsEngine)
add_subdirectory(LightYearsGame)
