function(CopyDllDirectoryFromLibToTarget LIB_NAME TARGET_NAME)
add_custom_command(
	TARGET ${TARGET_NAME}
	POST_BUILD
	COMMAND
	${CMAKE_COMMAND} -E copy_directory
	# TARGET FILE indicates the build file
	$<TARGET_FILE_DIR:${LIB_NAME}>
	$<TARGET_FILE_DIR:${TARGET_NAME}>
)
endfunction()

COPYDLLDIRECTORYFROMLIBTOTARGET(${PROJECT_ENGINE_TARGET_NAME} ${PROJECT_TARGET_NAME})
